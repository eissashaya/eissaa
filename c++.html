أساسيات لغة C++

مقدمة عن C++

لغة C++ هي لغة برمجة قوية تُستخدم في تطوير البرمجيات عالية الأداء مثل الألعاب، أنظمة التشغيل، والتطبيقات المعقدة. تتميز بدعم البرمجة الكائنية (OOP) وسرعة التنفيذ.

كتابة أول برنامج

#include <iostream>
using namespace std;
int main() {
    cout << "مرحباً بك في C++!";
    return 0;
}

المتغيرات وأنواع البيانات

  • int: الأعداد الصحيحة
  • float, double: الأعداد العشرية
  • char: حرف واحد
  • string: نص
  • bool: منطقي (صحيح/خطأ)
int age = 20;
double pi = 3.14;
char grade = 'A';
bool isStudent = true;

جمل الإدخال والإخراج

int x;
cout << "ادخل رقم: ";
cin >> x;
cout << "الرقم هو: " << x;

الشروط (if - else)

int num = 10;
if(num > 5) {
    cout << "أكبر من 5";
} else {
    cout << "5 أو أقل";
}

الحلقات (for - while)

for(int i=0; i<5; i++) {
    cout << i << " ";
}
// حلقة while
int j = 0;
while(j < 5) {
    cout << j << " ";
    j++;
}

الدوال (Functions)

int sum(int a, int b) {
    return a + b;
}
int main() {
    cout << sum(3, 4);
}

المصفوفات (Arrays)

int arr[3] = {1, 2, 3};
cout << arr[0]; // يطبع 1

البرمجة الكائنية (OOP) - تعريف كلاس

class Person {
public:
    string name;
    int age;
    void sayHello() {
        cout << "مرحباً!";
    }
};
